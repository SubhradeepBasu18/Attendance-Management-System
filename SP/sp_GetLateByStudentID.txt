ALTER PROC [dbo].[sp_GetLateByStudentID]
    @id INT  -- Hash ID
AS
BEGIN
    SET NOCOUNT ON;

    SELECT COUNT(*) AS LateCount
    FROM attendance at
    INNER JOIN hashtable ht 
        ON at.studentID = ht.studentID
    WHERE ht.id = @id
      AND at.status = 'LATE';
END

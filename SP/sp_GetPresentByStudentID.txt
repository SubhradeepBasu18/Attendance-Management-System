ALTER PROC [dbo].[sp_GetPresentByStudentID]
    @id INT  -- Hash ID
AS
BEGIN
    SET NOCOUNT ON;

    SELECT COUNT(*) AS PresentCount
    FROM attendance at
    INNER JOIN hashtable ht 
        ON at.studentID = ht.studentID
    WHERE ht.id = @id
      AND at.status = 'PRESENT';
END
